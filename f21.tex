\section{Line Integrals, Parametrization and Vector Fields}
\subsection{Identities between operations}

We have seen three operations defined by \( \nabla \).

\begin{align*}
    & \text{Gradient: } \nabla f && \text{Divergence: } \nabla \cdot  \vec{F} &&& \text{Curl: } \nabla \times \vec{F}
\end{align*}

There are many of them, we look at only one.

\begin{prop}
    For any scalar field \( f \) we have

    \[ \nabla \times (\nabla f) = 0 \]
\end{prop}

\begin{proof}
    We have \( \nabla f = \left( f_{x}, f_{y}, f_{z} \right) \). Then

    \[    \nabla \times \vec{F} =
    \left|\begin{bmatrix}
    \vec{i} & \vec{j} & \vec{k} \\
    \partial_x & \partial_y & \partial_z \\
    f_{x} & f_{y} & f_{z}
\end{bmatrix}\right| = \vec{i}\left(f_{zy} - f_{yz}, -f_{zx} + f_{xz}, f_{yx} - f_{cy}  \right) \]

But partial derivatives can be exhanged, \( f_{xy} = f_{yx} \). Then we find that

\[ \nabla \times \left( \nabla f \right) = 0 \]
\end{proof}

We are going to use this when we discuss conservative fields.

\subsection{Line Integrals}

Consider the curve C with

\[ \vec{r}(t) = (x(t), y(t)), \quad t_{0} \leq t \leq t_{1}\]

Consider the points \( \vec{r}(t) \) and \( \vec{r}(t + \Delta t) \) where \( \Delta t \) is small.

\[ \Delta x = x \left( t + \Delta t \right) - x \left( t \right), \quad \Delta y = y \left( t + \Delta t \right) - y \left( t \right) \]

The distance between the two points is

\[ \Delta S = \sqrt{\left( \Delta x \right)^2 + \left( \Delta y  \right)^2} \]

When \( \Delta t \) is small, we can use the following approximation

\[ \Delta x \approx \frac{d x}{dt} \Delta t, \quad \Delta y \approx \frac{d y}{dt} \Delta t\]

Then, for the distance, we get

\[ S = \sqrt{\left( \frac{d x}{dt}\Delta t \right)^2 + \left( \frac{d y}{dt} \Delta t \right)^2}  = \Delta t \sqrt{\left( \frac{d x}{dt} \right)^2 + \left( \frac{d y}{dt} \right)^2}\]

\begin{definition}
    The line integral of \( f \left( x, y \right)\) along a curve \( C \) is defined by

    \[ \int_{C} fdS = \int_{t_{0} }^{t_{1} } f \left( x \left( y \right), y \left( t \right) \right) \sqrt{x' \left( t \right)^2 + y' \left( t \right)^2} dt \]
\end{definition}

Some observations:

\begin{itemize}
    \item Note that \( f \) is restricted to \( \vec{r} \left( t \right) = \left( x \left( t \right), y \left( t \right) \right) \).
    \item When \( f = 1 \), we recover the arc length.

\end{itemize}

\begin{eg}
    Consider \( C \) definet by

    \[ x \left(  t \right), y \left( t \right) = 0, \quad 0 \leq t \leq 1 \]

    First, we compute

    \[ \sqrt{x' \left( t \right)^2 + y' \left( t \right)^2} dt = \sqrt{1^2+0^2} = 1 \]
\end{eg}

Now, consider \( f \left( x, y \right) =x^2+y \). Compute \( \int_{C}fdS \). Restricting \( f \) to \( C \) gives

\[ f \left( x \left( t \right), y \left( t \right) \right) = x \left( t \right)^2 + y \left( t \right) = t^2 + 0 =t^2 \]

Then we obtain

\[ \int_{C} fdS = \int_{0}^{1}  t^2 \cdot 1 dt = \frac{1}{3} \]

\begin{itemize}
    \item Line integrals can be used to compute the mass of an objects (1 - dimensional)
    \item The curve \( C \) describes the object
    \item The function \( \int_{C}fdS \) is the mass.
\end{itemize}

\subsection{Parametrization and orientation}

Next results as for the arc legnth.

\begin{prop}
    The integral \( \int_{C}fdS \) does not depend on the parametrization of \( C \).
\end{prop}

We will consider a special case.

\begin{eg}
    Consider \( C \) with

    \[ \vec{r} \left( t \right) = \left( x \left( t \right), y \left( t \right) \right), \quad t_{0} \leq t \leq t_{1}  \]

    \[ \text{Start: A} = \left( x \left( t_{0}  \right), y \left( t_{0}  \right) \right), \quad \text{End: B} = \left( x \left( t0 \right), y \left( t_{1}  \right) \right)\]

    We want to go from B to A, which can be done in the following manner

    \[ \vec{r}_{\text{opp}} = \left( x \left( -t \right), y \left( -t \right) \right), \quad -t_{1} \leq t \leq -t_{0}   \]

    Note that

    \[ \vec{r}_{\text{opp}}\left( -t_{1}  \right) = \left( x\left( t_{1}  \right), y\left( t_{1}  \right) \right) = \text{B} \]

    \[ \vec{r}_{\text{opp}}\left( -t_{0}  \right) = \left( x\left( t_{0}  \right), y\left( t_{0}  \right) \right)       = \text{A} \]
\end{eg}

\begin{eg}
    Consider \( C = \) segement from (0,0), to (1,0), take \( f\left( x,y \right) = x \). Show that \( \int_{C}fdS = \frac{1}{2}  \) using \( \vec{r}\left( t \right)\) and \(\vec{r}_{\text{opp}}\left( t \right) \).
\end{eg}

    If \( C \) parametrized by \( \vec{r}\left( t \right) \), we will use \( -C \) when considering \( \vec{r}_{\text{opp}}\left( t \right) \). We have that

    \[ \int_{C}fdS = \int_{-C}fdS \]

    The situation will be different for vector fields.

\subsection{Case of Vector Fields}

Consider the curve with

\[ \vec{r}\left( t \right)=\left( x\left( t \right), y\left( t \right) \right), \quad t_{0} \leq t \leq t_{1}  \]

\[ \vec{r'}\left( t \right) = \left( x'\left( t \right), y'\left( t \right) \right) \quad \textit{Velocity Vector}\]

\begin{definition}
    The line integral of \( \vec{F} \) along \( C \) is

    \[ \int_{C}\vec{F}\cdot d\vec{r} = \int_{t_{0} }^{t_{1} } \vec{F}\left( \vec{r}\left( t \right) \right)\cdot  \vec{r'}\left( t \right) dt\]

    Here \( \vec{F}\left( \vec{r}\left( t \right) \right) = \vec{F}\left( x\left( t \right), y\left( t \right) \right) \)
\end{definition}

\begin{itemize}
    \item In physics, we have that \( \vec{F} = \text{Force} \) and \( \int_{C}\vec{F}d \vec{r} = \) Work done by \( \vec{F} \) along \( C \).
\item The elementary case \( W = FS \), where \( W \) is work, \( F \) is Force and \( S \) is the displacement.
\end{itemize}

More explicitly, we consider

\[ \vec{F}\left( x,y \right) = \left( P\left( x,y \right), Q\left( x,y \right) \right) \]

Then we have that

\[ \vec{F}\left( \vec{r}\left( t \right)\cdot \vec{r'}\left( t \right) \right) = P\left( x\left( t \right), y\left( t \right) \right)\cdot x'\left( t \right) + Q\left( x\left( t \right), y\left( t \right) \right)\cdot y'\left( t \right) \]

The line integral is then

\[ \int_{C} \vec{F} \cdot d \vec{r} = \int_{t_{0} }^{t_{1} } P\left( x\left( t \right),y\left( t \right) \right)\cdot x'\left( t \right) dt + \int_{t_{0} }^{t_{1} } Q\left( x\left( t \right),y\left( t \right) \right)\cdot y'\left( t \right) dt \]      

We will write the expression as

\[ \int_{C}\vec{F}d \vec{r} = \int_{C}P\ dx + \int_{C}Q\ dy \]

\begin{eg}
    Consider a curve \( C \) with

    \[ \vec{r}\left( t \right) = \left( t, t^2 \right),\quad 0 \leq t \leq 1\]

    We have \( x\left( t \right) = t \) and \( y\left( t \right)=t^2 \), its derivative is

    \[ \vec{r'}\left( t \right)= \left( x'\left( t \right), y'\left( t \right) \right)= \left( 1,2t \right) \]
        
    Now consider the vector field

    \[ \vec{F}\left( x,y \right) = \left( x+y,x \right)\]

    That is \( P = x+ y \text{ and } Q = x \). When restricted to \( C \), we get

    \[ \vec{F}\left( \vec{r}\left( t \right) \right) = \left( x\left( t \right)+ y\left( t \right), y\left( t \right) \right) = \left( t+t^2, t \right)\]

    We want to compute

    \[ \vec{F}\left( \vec{r}\left( t \right) \right) \cdot \vec{r'}\left( t \right) = \left( t+t^2, t \right) \cdot  \left( 1,2t \right) = \left( 3t^2 + t \right) \]

    We then obation the following

    \[ \int_{C}\vec{F} \cdot  d \vec{r} = \int_{0}^{1} \left( 3t^2 + t \right) dt = \frac{3}{2}  \]
\end{eg}
